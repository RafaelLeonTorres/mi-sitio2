// Ejemplo de Árbol Binario en C#
using System;

class TreeNode
{
    public int Value;
    public TreeNode Left, Right;

    public TreeNode(int value)
    {
        Value = value;
        Left = Right = null;
    }
}

class Program
{
    static void Main()
    {
        TreeNode root = new TreeNode(10);
        root.Left = new TreeNode(5);
        root.Right = new TreeNode(15);

        Console.WriteLine($"Root: {root.Value}");
        Console.WriteLine($"Left Child: {root.Left.Value}");
        Console.WriteLine($"Right Child: {root.Right.Value}");
    }
}

// Ejemplo de Stack en C#
using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        Stack stack = new Stack();
        stack.Push("First");
        stack.Push("Second");
        stack.Push("Third");

        while (stack.Count > 0)
        {
            string item = stack.Pop();
            Console.WriteLine(item);
        }
    }
}

// Ejemplo de Queue en C#
using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        Queue queue = new Queue();
        queue.Enqueue("Task 1");
        queue.Enqueue("Task 2");
        queue.Enqueue("Task 3");

        while (queue.Count > 0)
        {
            string task = queue.Dequeue();
            Console.WriteLine(task);
        }
    }
}

// Ejemplo de ConcurrentQueue en C#
using System;
using System.Collections.Concurrent;

class Program
{
    static void Main()
    {
        ConcurrentQueue queue = new ConcurrentQueue();
        queue.Enqueue("Task 1");
        queue.Enqueue("Task 2");

        while (queue.TryDequeue(out string task))
        {
            Console.WriteLine(task);
        }
    }
}
// Ejemplo de ConcurrentDictionary en C#
using System;
using System.Collections.Concurrent;

class Program
{
    static void Main()
    {
        ConcurrentDictionary dict = new ConcurrentDictionary();
        dict.TryAdd("Alice", 30);
        dict.TryAdd("Bob", 25);

        foreach (var pair in dict)
        {
            Console.WriteLine($"{pair.Key}: {pair.Value} años");
        }
    }
}

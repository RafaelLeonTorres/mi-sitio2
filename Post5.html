// C# Example for GET method in ASP.NET Core 8
[ApiController]
[Route("api/[controller]")]
public class UsersController : ControllerBase
{
    private readonly List _users = new List
    {
        new User { Id = 1, Name = "Alice" },
        new User { Id = 2, Name = "Bob" }
    };
    
    [HttpGet]
    public ActionResult> Get()
    {
        return Ok(_users);
    }
}

public class User
{
    public int Id { get; set; }
    public string Name { get; set; }
}

// C# Example for POST method in ASP.NET Core 8
[HttpPost]
public IActionResult Post([FromBody] User user)
{
    _users.Add(user);
    return CreatedAtAction(nameof(Get), new { id = user.Id }, user);
}

// C# Example for PUT method in ASP.NET Core 8
[HttpPut("{id}")]
public IActionResult Put(int id, [FromBody] User user)
{
    var existingUser = _users.FirstOrDefault(u => u.Id == id);
    if (existingUser == null)
    {
        return NotFound();
    }

    existingUser.Name = user.Name;
    return NoContent();
}

// C# Example for DELETE method in ASP.NET Core 8
[HttpDelete("{id}")]
public IActionResult Delete(int id)
{
    var user = _users.FirstOrDefault(u => u.Id == id);
    if (user == null)
    {
        return NotFound();
    }

    _users.Remove(user);
    return NoContent();
}

// C# Example for Pagination in ASP.NET Core 8
[HttpGet]
public ActionResult> Get(int pageNumber = 1, int pageSize = 10)
{
    var pagedUsers = _users.Skip((pageNumber - 1) * pageSize).Take(pageSize);
    return Ok(pagedUsers);
}
